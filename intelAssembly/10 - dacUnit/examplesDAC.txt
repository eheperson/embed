;---------------------------------------------------------------------------
; dac 4. ÖRNEK
; asagida DACCON ayarinda harici ref seçilmis
; Sekronizasyon modu kullanilmis sync=0 

;SYNC Temizlendiginde, her iki DAC senkron calisma moduna girer. Bu modda, öncelikle
;DACiL/H kaydedicilerine veri aktarilir; daha sonra da SYNC biti lojik ‘1’yapilarak her
;iki DAC çikisi ayni anda güncellenmis olur.

; 1 Voltluk ve 0.5 Voltluk çikis ayarlanmis

; DAC  12 bit olarak ayarlanmis = 12 bit 4095d degerindedir.
; V_harici_ref=4 Volt, 4095-->> 4 Volt olur

; 1   Volt için 4095/4=1024 DAC0L/DAC0H yüklenmesi gerekir.
; 0.5 Volt için 4095/8=512d DAC1L/DAC1H yüklenmesi gerekir.

	org 	0000h
BASLA:
	MOV 	DACCON, #01111011b  ; mode=0 12 bit -- rng1=1(Harici REF) -- rng0=1(Harici REF) -- clr1=1 --  clr0=1 --  sync=0 -- pd1=1 --  pd0=1(enerji)
	MOV		DPTR,#1024D
	MOV		DAC0H,DPH
	MOV 	DAC0L,DPL
	
	MOV		DPTR,#512D
	MOV		DAC1H,DPH
	MOV		DAC0L,DPL
	
	ORL		DACCON,#00000100b;  sync=1 yapildigi anda cikislar guncellenir.
	x:		sjmp	x

END
